nuclear_test_detected.1 = {
	type = country_event

	title = nuclear_test_detected.1.t
	desc = nuclear_test_detected.1.d
	flavor = nuclear_test_detected.1.f

	trigger = {
		has_law = law_nuclear_proliferation
		any_scope_state = {
			has_building = building_uranium_mine
		}
		any_scope_state = {
			has_building = building_nuclear_power_plant
		}
		NOT = { has_modifier = modifier_nuclear_arsenal }
	}

	option = {
		name = nuclear_test_detected.1.a
		effect_nuclear_test = yes
		every_country = {
			limit = {
				country_rank >= rank_value:major_power
				NOT = { THIS = ROOT }

			trigger_event = { id = nuclear_crisis_response.1 }
		}
	}

	option = {
		name = nuclear_test_detected.1.b
		change_prestige = -5
		add_modifier = {
			name = modifier_digital_economy_bonus

		}
	}

	option = {
		name = nuclear_test_detected.1.c
		change_prestige = -15
		change_infamy = -5
	}
}
