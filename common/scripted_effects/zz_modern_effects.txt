# Modern Scripted Effects — vic3-modern-2000

effect_impose_sanctions = {
    $TARGET$ = {
        add_modifier = {
            name = modifier_sanctions_target
            months = 36
        }
    }
    add_modifier = {
        name = modifier_sanctions_sender
        months = 36
    }
}

effect_lift_sanctions = {
    $TARGET$ = {
        remove_modifier = modifier_sanctions_target
    }
    remove_modifier = modifier_sanctions_sender
}

effect_cyber_attack = {
    $TARGET$ = {
        add_modifier = {
            name = modifier_cyber_attack_victim
            months = 12
        }
    }
}

}
